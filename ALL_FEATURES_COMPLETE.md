## ğŸ› ï¸ How to Use These Features

### 1. Integrate Dark Mode
```tsx
import { DarkModeToggle } from '@/components/ui/DarkModeToggle';

// Add to your header:
<DarkModeToggle />
```

### 2. Add Command Palette
```tsx
import { CommandPalette } from '@/components/ui/CommandPalette';

// Add to root layout:
<CommandPalette />
```
Then press **Cmd+K** (Mac) or **Ctrl+K** (Windows)!

### 3. Display Cost Calculator
```tsx
import { CostSavingsCalculator } from '@/components/analytics/CostSavingsCalculator';

<CostSavingsCalculator 
  documentsGenerated={documents.length}
  documentsSaved={savedDocs.length}
/>
```

### 4. Show Analytics Dashboard
```tsx
import { AdvancedAnalyticsDashboard } from '@/components/analytics/AdvancedAnalyticsDashboard';

<AdvancedAnalyticsDashboard documents={allDocuments} />
```

### 5. Add Document Search
```tsx
import { DocumentSearch } from '@/components/documents/DocumentSearch';

<DocumentSearch 
  documents={documents}
  onResultsChange={(filtered) => setFilteredDocs(filtered)}
/>
```

### 6. Enable Tags
```tsx
import { DocumentTagManager } from '@/components/documents/DocumentTagManager';

<DocumentTagManager 
  documentId={doc._id}
  initialTags={doc.tags}
  onTagsUpdate={(tags) => updateTags(doc._id, tags)}
/>
```

### 7. Show Folders
```tsx
import { DocumentFolders } from '@/components/documents/DocumentFolders';

<DocumentFolders 
  currentFolder={selectedFolder}
  onFolderChange={(folder) => setSelectedFolder(folder)}
/>
```

### 8. Display Version History
```tsx
import { VersionHistory } from '@/components/documents/VersionHistory';

<VersionHistory 
  documentId={doc._id}
  currentVersion={currentVersion}
  versions={previousVersions}
  onRestore={(versionId) => restoreVersion(versionId)}
/>
```

### 9. AI Analyzer (Requires OpenAI API)
```tsx
import { AIDocumentAnalyzer } from '@/components/ai/AIDocumentAnalyzer';

<AIDocumentAnalyzer 
  documentContent={doc.content}
  documentType={doc.type}
  onAnalysisComplete={(result) => showResults(result)}
/>
```

### 10. Comments (Requires Pusher/Socket.io)
```tsx
import { DocumentComments } from '@/components/collaboration/DocumentComments';

<DocumentComments 
  documentId={doc._id}
  currentUser={user}
  onAddComment={(content) => addComment(content)}
/>
```
